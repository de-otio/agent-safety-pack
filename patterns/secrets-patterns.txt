# Secret and credential detection patterns
# One extended regex per line. Lines starting with # are comments.
# Used by PostToolUse hooks to scan command output, file content,
# and other tool results for accidentally exposed secrets.

# ============================================================
# CLOUD PROVIDER KEYS
# ============================================================

# AWS Access Key ID
(A3T[A-Z0-9]|AKIA|AGPA|AIDA|AROA|AIPA|ANPA|ANVA|ASIA)[A-Z0-9]{16}
# AWS Secret Access Key (40-char base64)
aws_secret_access_key\s*[:=]\s*[A-Za-z0-9/+=]{40}
# AWS Session Token
aws_session_token\s*[:=]\s*[A-Za-z0-9/+=]{100,}
# Google API Key
AIza[0-9A-Za-z_-]{35}
# Google OAuth Client Secret
client_secret.*\.apps\.googleusercontent\.com
# Google Service Account Key
"type"\s*:\s*"service_account"
# Azure Storage Account Key
AccountKey=[A-Za-z0-9+/=]{88}
# Azure AD Client Secret
\b[a-zA-Z0-9~_.-]{34}\b.*\b(client.?secret|AZURE)\b
# DigitalOcean
\bdop_v1_[a-f0-9]{64}\b
\bdo[a-z]{2}_[a-zA-Z0-9]{43}\b
# Heroku API Key
\b[0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}\b.*heroku
heroku.*\b[0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}\b

# DATABASE AND INFRASTRUCTURE
# ============================================================

# Connection strings with embedded credentials

# ============================================================
# CRYPTOGRAPHIC MATERIAL
# ============================================================

# Private keys (PEM format)
# Pattern avoids leading dashes to prevent grep option injection on BSD grep
-{5}BEGIN\s+(RSA\s+|EC\s+|OPENSSH\s+|DSA\s+|ENCRYPTED\s+)?PRIVATE\s+KEY-{5}
# PGP private key block
-{5}BEGIN PGP PRIVATE KEY BLOCK-{5}
# Certificate (less sensitive but worth flagging in output)
-{5}BEGIN CERTIFICATE-{5}

# ============================================================
# JSON WEB TOKENS
# ============================================================

# JWT (three base64url segments separated by dots)
\beyJ[a-zA-Z0-9_-]{10,}\.eyJ[a-zA-Z0-9_-]{10,}\.[a-zA-Z0-9_-]{10,}\b

# ============================================================
# GENERIC HIGH-SIGNAL PATTERNS
# ============================================================

# Explicit secret assignment in config/env output
(?i)(api[_-]?key|api[_-]?secret|access[_-]?key|access[_-]?token|auth[_-]?token|secret[_-]?key|private[_-]?key|client[_-]?secret|app[_-]?secret|encryption[_-]?key|signing[_-]?key|bearer[_-]?token)\s*[:=]\s*['"]?[a-zA-Z0-9/+=_-]{16,}['"]?
# password in config/env output (with value, not just the word)
(?i)(password|passwd|pwd)\s*[:=]\s*['"]?[^\s'"]{8,}['"]?
# Bearer token in headers
(?i)Authorization:\s*Bearer\s+[a-zA-Z0-9._-]{20,}
# Basic auth header (base64 encoded credentials)
(?i)Authorization:\s*Basic\s+[A-Za-z0-9+/=]{10,}
# .env file content pattern (KEY=value with high-entropy value)
^[A-Z_]{3,50}=[a-zA-Z0-9/+=_-]{20,}$
